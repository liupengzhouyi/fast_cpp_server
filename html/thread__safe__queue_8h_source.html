56 bool push(const T& item, bool block = true) {
57 std::unique_lock<std::mutex> lock(mutex_);
58
59 if (!block) {
60 // 非阻塞模式，若队列已满则直接返回 false
61 if (queue_.size() >= max_size_) {
62 if (usedLog_) {
63 MYLOG_WARN("[ThreadSafeQueue::push] Non-blocking mode: queue is full.");
64 }
65 return false;
66 }
67 } else {
68 // 阻塞等待队列未满，避免使用 lambda，改为手动条件判断
69 while (queue_.size() >= max_size_ && !shutdown_) {
70 cond_not_full_.wait(lock); // 等待有空间
71 }
72 }
73
74 // 如果已经 shutdown，拒绝插入
75 if (shutdown_) {
76 if (usedLog_) {
77 MYLOG_WARN("[ThreadSafeQueue::push] Queue is shutdown, cannot push new items.");
78 }
79 return false;
80 }
81
82 // 插入元素
83 queue_.push_back(item);
84 if (usedLog_) {
85 MYLOG_INFO("[ThreadSafeQueue::push] Pushed item successfully, queue size: {}", queue_.size());
86 }
87 // 通知一个等待中的 pop() 操作
88 cond_not_empty_.notify_one();
89 return true;

98 bool pop(T& result, bool block = true, int timeout_ms = -1) {
99 std::unique_lock<std::mutex> lock(mutex_);
100
101 // 非阻塞模式
102 if (!block) {
103 if (queue_.empty()) {
104 if (usedLog_) {
105 MYLOG_WARN("[ThreadSafeQueue::pop] Non-blocking mode: queue is empty.");
106 }
107 return false;
108 }
109 }
110 else {
111 // 阻塞 + 无限等待
112 if (timeout_ms < 0) {
113 // 等待直到队列非空或 shutdown
114 while (queue_.empty() && !shutdown_) {
115 if (usedLog_) {
116 MYLOG_INFO("[ThreadSafeQueue::pop] queue is empty and not shutdown; Blocking wait (no timeout)...");
117 }
118 cond_not_empty_.wait(lock);
119 }
120 }
121 // 阻塞 + 超时等待
122 else {
123 auto now = std::chrono::steady_clock::now();
124 auto timeout_time = now + std::chrono::milliseconds(timeout_ms);
125
126 while (queue_.empty() && !shutdown_) {
127 if (usedLog_) {
128 MYLOG_INFO("[ThreadSafeQueue::pop] Blocking wait with timeout ({})...", timeout_ms);
129 }
130
131 // wait_until 返回 false 表示超时
132 if (cond_not_empty_.wait_until(lock, timeout_time) == std::cv_status::timeout) {
133 if (usedLog_) {
134 MYLOG_WARN("[ThreadSafeQueue::pop] Timeout reached while waiting for items.");
135 }
136 return false;
137 }
138 }
139 }
140 }
141
142
143 // 如果队列为空，并且已关闭，说明不会再有数据，返回失败
144 if (queue_.empty()) {
145 if (usedLog_) {
146 MYLOG_WARN("[ThreadSafeQueue::pop] Queue is empty after waiting, and shutdown is true.");
147 }
148 return false;
149 }
150
151 // 从队列中获取数据
152 result = std::move(queue_.front());
153 queue_.pop_front();
154
155 if (usedLog_) {
156 MYLOG_INFO("[ThreadSafeQueue::pop] Popped item successfully, queue size: {}", queue_.size());
157 }
158
159 // 通知可能在等待空间的 push 操作
160 cond_not_full_.notify_one();
161 return true;
162 }
